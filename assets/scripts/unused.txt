function renderSolid(x, y, w, h) {
  var left = isTouching(x - player.speed, y, player.speed, h);
  var top = isTouching(x, y - player.speed, w, player.speed);
  var right = isTouching(x + w, y, player.speed, h);
  var down = isTouching(x, y + h, w, player.speed);
  if (left == true && player.keys.right == true) {
    player.x = x - player.width;
  } else if (top == true && player.keys.down == true) {
    player.y = y - player.height;
  } else if (right == true && player.keys.left == true) {
    player.x = x + w;
  } else if (down == true && player.keys.up == true) {
    player.y = y + h;
  }
}
